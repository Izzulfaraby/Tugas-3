    FOR j FROM 1 TO (n - i) DO
        PRINT " "  // Print space for alignment
    END FOR
    FOR j FROM 1 TO i DO
        PRINT "*"  // Print star
        PRINT " "  // Print space after star
    END FOR
    PRINT NEWLINE  // Move to the next line
END FOR
END
